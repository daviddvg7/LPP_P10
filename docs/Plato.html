    # File lib/comida/plato.rb, line 62
def get_carbos()
    i=@alimentos.head
    f=@cantidades.head
    carbos_total=0
    cantidad_total=0

    loop do
        aux=(i.value).carbos*(f.value/100)
        carbos_total+=aux
        cantidad_total += f.value
        i=i.next
        f=f.next
    break if i==nil
    end
    carbos=((carbos_total/cantidad_total)*100).round(1)
    return "#{carbos}%"
end

def get_huella_energetica
    
    if (self.get_kcal < 670)
        return 1
    elsif (self.get_kcal > 830)
        return 3
    else
        return 2
    end
end

def to_s
    i=@alimentos.head
    f=@cantidades.head
    texto="El plato #{@nombre} está compuesto de "

    loop do

        if i.next==nil
            texto += "y " + (f.value).to_s + " g de " + (i.value).nombre + "."
        else
            texto += (f.value).to_s + " g de " + (i.value).nombre + ", "
        end
        
        i=i.next
        f=f.next
    break if i==nil
    end
    return texto
end
