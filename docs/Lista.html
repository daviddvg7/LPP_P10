def insert_head(valor)
    aux=Nodo.new(valor, nil, nil)
    if @head==nil
        @head=aux
        @tail=aux
    elsif @head==@tail
        @head.prev=aux
        @tail.prev=head
        aux.next=head
        @head=aux
    else
        @head.prev=aux
        aux.next=@head
        @head=aux
    end
end

def terreno()
    terreno=0
    aux=@head
    loop do
        terreno+=(aux.value).terreno
        aux=aux.next
    break if aux==nil
    end
    
    return terreno.round(1)
end
